\begin{tikzpicture}
    \def\dx{0}
    \foreach[evaluate={\dy=-\z; \xeval=5-\z}] \z in {0,...,4}{
        \ifnum \z<4
            \foreach \x in {2,...,\xeval}
                \draw [->,thick] (\x+\dx-0.9,\dy+0.1) .. controls (\x+\dx-0.5,\dy+0.5) .. (\x+\dx-0.1,\dy+0.1)
                node[pos=0.5,yshift=1ex]{\x};
        \fi
        \foreach \x in {1}
            \filldraw[fill=black] (\x+\dx,\dy) circle [radius=0.1];
        \foreach \x in {2,3,4,5}
            \filldraw[fill=white] (\x+\dx,\dy) circle [radius=0.1];
        \foreach[evaluate={\j=\x==1?\x:int(\x+1)}] \x in {\z,...,0}
            \draw [red,<-,thick] (5.1-\x+\dx,\dy-0.1) .. controls (5.5-\x+\dx,\dy-0.5) .. (5.9-\x+\dx,\dy-0.1)
            node[pos=0.5,yshift=-1ex]{\j};
        }

\end{tikzpicture}
